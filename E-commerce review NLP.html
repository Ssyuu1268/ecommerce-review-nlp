# These are negation or contrastive words. 
# They carry important sentiment or semantic meaning and should NOT be removed as regular stopwords.
NEGATION_WORDS = {
    "not", "no", "never", "nothing", "nowhere", "hardly", "scarcely", "barely",
    "dont", "isnt", "didnt", "wasnt", "cant", "wont", "however", "but"
}

# Custom stopword list (excluding negation words)
custom_stop_words = set(ENGLISH_STOP_WORDS) - NEGATION_WORDS

# Text preprocessing function (supports negation emphasis and optional lemmatization)
def preprocess_text(text, 
                    remove_stopwords=True, 
                    use_lemmatization=True,
                    emphasize_negation=True):
    # Lowercase + remove punctuation and symbols
    text = str(text).lower()
    text = re.sub(r"[^a-z\s]", "", text)       # Keep only letters
    text = re.sub(r"\s+", " ", text).strip()   # Remove extra whitespaces

    # Tokenization
    tokens = text.split()

    # Remove stopwords (but keep negation words)
    if remove_stopwords:
        tokens = [word for word in tokens if word not in custom_stop_words]

    # Lemmatization (e.g., loved  love)
    if use_lemmatization:
        lemmatizer = WordNetLemmatizer()
        tokens = [lemmatizer.lemmatize(w) for w in tokens]

    # Emphasize negation (e.g., not  NOT_not)
    if emphasize_negation:
        tokens = ["NOT_" + w if w in NEGATION_WORDS else w for w in tokens]

    return " ".join(tokens)

X_test_pad = pad_sequences(X_test_seq, maxlen=max_len)

model_cnn.add(Conv1D(128, 5, activation='relu'))            # Convolution layer
model_cnn.add(MaxPooling1D(pool_size=2))
model_cnn.add(Conv1D(64, 5, activation='relu'))
model_cnn.add(GlobalMaxPooling1D())                         # Global pooling
model_cnn.add(Dense(64, activation='relu'))
model_cnn.add(Dense(5, activation='softmax'))               # Output layer (5 classes)

    optimizer='adam',

    patience=2,
    restore_best_weights=True

from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score, confusion_matrix

def evaluate_model(name, y_true, y_pred):
    accuracy = accuracy_score(y_true, y_pred)
    precision = precision_score(y_true, y_pred, average='weighted', zero_division=0)
    recall = recall_score(y_true, y_pred, average='weighted', zero_division=0)
    f1 = f1_score(y_true, y_pred, average='weighted', zero_division=0)
    cm = confusion_matrix(y_true, y_pred)
    
    print(f"\n=== {name} ===")
    print(f"Accuracy : {accuracy:.3f}")
    print(f"Precision: {precision:.3f}")
    print(f"Recall   : {recall:.3f}")
    print(f"F1-Score : {f1:.3f}")
    
    # Plot confusion matrix
    ConfusionMatrixDisplay.from_predictions(y_true, y_pred, cmap='Blues')
    plt.title(f"{name} - Confusion Matrix")
    plt.show()
    
    return {
        "Model": name,
        "Accuracy": round(accuracy, 3),
        "Precision": round(precision, 3),
        "Recall": round(recall, 3),
        "F1-Score": round(f1, 3)
    }

import time

# Measure prediction time for a model
def timed_prediction(model, X_input, is_neural=False):
    start = time.time()
    
    if is_neural:
        y_pred = model.predict(X_input)
        y_pred = np.argmax(y_pred, axis=1) + 1  # # Neural model: convert probabilities to class labels (15)
    else:
        y_pred = model.predict(X_input)
        
    end = time.time()
    elapsed = round(end - start, 3)
    return y_pred, elapsed

# Preprocess input text (for CNN / LSTM models)
def preprocess_input(text, tokenizer, max_len=200):
    seq = tokenizer.texts_to_sequences([text])
    return pad_sequences(seq, maxlen=max_len)

# General prediction function: supports four model types
def predict_product_rating(text, model_filename):
    if model_filename.endswith(".pkl"):
        # Traditional ML models: Naive Bayes / KNN
        model = joblib.load(model_filename)
        return int(model.predict([text])[0])

    elif model_filename.endswith(".keras"):
        # Deep learning models: CNN / LSTM
        model_type = model_filename.split("_")[0]    # 'cnn' or 'lstm'
        tokenizer_path = f"tokenizer_{model_type}.pkl"

        # Load tokenizer
        with open(tokenizer_path, "rb") as f:
            tokenizer = pickle.load(f)

        # Preprocess input and load model
        x_input = preprocess_input(text, tokenizer)
        model = load_model(model_filename)
        prob = model.predict(x_input)

        # Choose the class with highest probability (+1 to match label range 15)
        return int(np.argmax(prob, axis=1)[0] + 1)

    else:
        raise ValueError("Unsupported model file type: " + model_filename)

Written Report

Introduction of the Dataset

The dataset comprises 59,621 entries featuring two critical columns: Score and Text, which represent product review ratings and the corresponding feedback. While it holds significant potential for text classification tasks, it presents several challenges that must be addressed for optimal results.

Firstly, it's important to note that nearly 7.8% of the Score values and 7.7% of the Text entries are missing, highlighting the urgent need for diligent data cleaning before any modeling can take place. Additionally, the dataset suffers from a marked class imbalanceover 58% of ratings are 5-starleaving 1-star and 2-star reviews severely underrepresented. This skew not only risks biasing models towards positive predictions but also diminishes the overall robustness of the analysis, unless strategies such as resampling or weighted class methods are employed.

Moreover, the variability inherent in user-generated contentranging from differences in text length to the use of informal languagecoupled with potential noise from typos or irrelevant information, further complicates the task at hand. These factors reinforce the necessity for meticulous preprocessing and thoughtful handling to ensure that model performance is strong and fair across all rating categories.

Addressing these issues not only enhances the quality of the analysis but also ensures more equitable outcomes across diverse reviews, ultimately leading to more reliable insights.

Data Preparation

Effected data preparation is essential for building robust and accurate text classification models. This project carefully applied several preprocessing steps to clean, normalize, and transform the review text data into formats suitable for both traditional machine learning and deep learning models.

The text data was first standardized by converting all characters to lowercase and removing unwanted characters such as punctuation, digits, and excessive whitespace. This ensured uniformity and reduced noise in the input features.Stopwords were removed to eliminate commonly occurring but semantically weak words (e.g., "the", "and" and "is"). However, negation-related words (e.g., "not", "never", "but") were explicitly retained and emphasized using a custom prefix (e.g., "NOT_not"). This enhancement was important for preserving sentiment-reversing constructs that strongly influence meaning. Lemmatization was applied using WordNet, reducing inflected words to their base forms (e.g., "running" "run"), thereby minimizing vocabulary size and improving generalization.

Next, the cleaned text was tokenized. For traditional models, theTfidfVectorizerwas used with a unigram and bigram range, enabling the model to learn not only individual word importance but also short phrase patterns. A maximum of 20,000 features was specified to maintain computational efficiency while capturing primary information. For deep learning models such as CNN and LSTM, Keras'sTokenizerconverted text into sequences of integers based on word indices. These sequences were then padded to a fixed length (200 tokens) to ensure uniform input dimensions, which is required for neural networks.

The target labels (scores ranging from 1 to 5) were one-hot encoded to match the format expected by categorical loss functions in deep learning. Furthermore, due to class imbalance (e.g., overrepresentation of score 5 and underrepresentation of scores 1 and 2), class weights were computed and applied during LSTM training to ensure fair learning across all classes.

These preprocessing steps were carefully selected and justified based on best practices in natural language processing. They collectively enhanced the model's ability to learn meaningful patterns, handle noisy or imbalanced data, and generalize well to unseen text (Kusumaningrum & Nugroho, 2017).

Machine Learning Method

This project explores four different machine learning architectures for multi-class text classification: Complement Naive Bayes (CNB), k-Nearest Neighbors (KNN), Convolutional Neural Networks (CNN), and Long Short-Term Memory (LSTM) networks. All models use either TF-IDF or word embeddings as the foundation for text representation, enabling them to process and learn from textual input effectively.

The Complement Naive Bayes (CNB) model is implemented within a TF-IDF-based pipeline to overcome the limitations of traditional Naive Bayes, particularly its sensitivity to class imbalance and strong feature independence assumptions. TF-IDF represents text with unigrams and bigrams, while rare terms are removed using a minimum document frequency threshold (Rennie et al., 2003).

The k-Nearest Neighbors (KNN) model utilizes TF-IDF features and classifies texts based on the majority class of the nearest neighbours in vector space. This architecture closely follows the methodology proposed by (Trstenjak et al., 2014), who demonstrated the effectiveness of combining KNN with TF-IDF representations for textual data. Several values ofkare tested through cross-validation, with the best result achieved atk = 9. The final accuracy reached approximately 63.5%, making KNN a simple and interpretable model, though less robust when dealing with sparse or imbalanced data.

The Convolutional Neural Networks (CNN) model starts by converting text into padded sequences of tokens, followed by a word embedding layer that maps each token to a dense 128-dimensional vector. Convolutional and max-pooling layers are applied to capture local semantic patterns, with global max-pooling used to reduce the feature map before classification. The CNN model achieved 70% accuracy and outperformed both CNB and KNN, particularly in its ability to detect minority class patterns. It also benefits from its strength in identifying structural features within short text (Kim, 2014).

The Long Short-Term Memory (LSTM) model processes tokenized and padded text sequences using stacked LSTM layers, allowing it to capture sequential and contextual relationships between words. A dense layer and dropout are included to improve generalization and reduce overfitting. Initially, during training, the model exhibited biased behaviour due to a strong class imbalance in the dataset, where a large portion of samples was labelled as score 5. This led the model to predict a score of 5 for most inputs, regardless of actual content, resulting in poor performance and lack of sensitivity to other classes. To correct this, class weights were computed and incorporated into the training process, assigning higher importance to underrepresented classes. This allowed the model to better learn from all score categories and reduced its over-reliance on the dominant class. Early stopping was also used to retain the best weights and avoid overfitting. After applying these techniques, the model achieved an accuracy of 65%, and in the final integrated system, it demonstrated a significantly improved ability to interpret input text and assign appropriate categories, reflecting a more balanced and effective learning outcome (Hochreiter & Schmidhuber, 1997).

Each model was implemented using Keras or Scikit-learn. For the neural models, padding length was fixed at 200 tokens, and embedding layers were initialized with a vocabulary size of 20,000 and output dimensions of 128. The CNN and LSTM architectures used ReLU activation and categorical cross-entropy loss, optimized using the Adam optimizer (Kingma & Ba, 2015). Early stopping with a patience of 2 was used to prevent overfitting by halting training once the validation performance stopped improving. In the TF-IDF-based pipelines, the n-gram range was set to (1,2) and min_df = 2 to filter out noise while preserving functional co-occurrence patterns. These text representation choices helped enhance the model's ability to capture semantic meaning and improve generalization.

In addition to designing and training the models, the system also includes a set of utility functions to evaluate model performance and runtime efficiency. Each models predictions were assessed using weighted accuracy, precision, recall, and F1-score, along with confusion matrix visualization. A timing function was also implemented to compare prediction latency across models, revealing trade-offs between accuracy and speedwhere Naive Bayes was the fastest and CNN achieved the highest F1 score.

A unified prediction interface was developed to support all four model types and improve usability and deployment readiness. Depending on the file format, the system automatically loads the appropriate model and tokenizer, preprocesses user input, and returns a predicted rating. An interactive command-line loop enables real-time testing of all models on arbitrary review inputs. These design choices strongly focus on practical evaluation, model comparison, and extensibility of the overall text classification framework.

Experimental Results

Four modelsNaive Bayes, K-Nearest Neighbors (KNN), Convolutional Neural Network (CNN), and Long Short-Term Memory (LSTM)were evaluated using accuracy, precision, recall, and F1-score. Among them,CNN achieved the best overall performance, with the highest accuracy (70.0%) and a strong F1-score (0.669), especially excelling in identifying high-rated reviews (class 5).Naive Bayesfollowed closely with 68.5% accuracy and showed excellent recall on the extreme classes (1 and 5), making it a strong and lightweight baseline. LSTMreached 65.0% accuracy and performed relatively well on middle-range scores, benefiting from its ability to capture context and the use of class weights. In contrast,KNNperformed the weakest (63.5% accuracy), struggling particularly with ambiguous classes due to its sensitivity to high-dimensional TF-IDF input.

Inference time also varied significantly. Naive Bayes was the fastest, while KNN was the slowest due to its non-parametric nature. CNN and LSTM offered a good balance between performance and efficiency, making them suitable for deployment. Overall, CNN proved to be the most effective model for this multi-class text classification task, while Naive Bayes remained a strong alternative for fast and reasonably accurate predictions.

Discussion

The overall results highlight that CNN achieved the best balance of accuracy and F1-score, while Naive Bayes performed well as a fast and lightweight baseline. However, several limitations should be noted. First,class imbalancenegatively impacted predictions for minority classes, particularly scores 2 and 3, which consistently showed lower recall and F1 scores. Although class weighting helped slightly in the LSTM model, it did not fully resolve the issue. Second, models oftenconfused adjacent ratings, such as predicting 4 as 5, familiar in ordinal classification where class boundaries are subtle and subjective.

In terms of usage, Naive Bayes is recommended for real-time or low-resource environments, while CNN or LSTM are more suitable when higher accuracy is required. Future improvements may include data augmentation, ordinal-aware loss functions, or pre-trained embeddings to enhance semantic understanding.
